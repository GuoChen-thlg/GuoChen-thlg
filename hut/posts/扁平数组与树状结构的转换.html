const arrayToTree = (arr, id, pid) => {
	if (!id || id == '') {
		return []
	}
	if (arr instanceof Array) {
		let tree = []
		let treeMap = {}
		arr.forEach(o => {
			treeMap[o[id]] = o
		})
		arr.forEach(o => {
			let parent = treeMap[o[pid]]
			if (parent && o[id] != o[pid]) {
				if (!parent.children) {
					parent.children = []
				}
				parent.children.push(o)
			} else {
				tree.push(o)
			}
		})
		return tree
	} else {
		return [arr]
	}
}
